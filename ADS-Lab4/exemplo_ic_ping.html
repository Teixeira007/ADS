library(ggplot2)
z_f <- function(confianca) {
  qnorm(1 - (1 - confianca)/2)
}

n_f <- function(z, desvio_padrao, margem_erro, media) {
  ((100 * z * desvio_padrao) / (margem_erro * media))^2
}

ic_erro_f <- function(z, desvio_padrao, tamanho_amostra) {
  z * (desvio_padrao / sqrt(tamanho_amostra))
}
arquivos <- c("ping_facebook_30.txt_filt", "ping_google_30.txt_filt", "ping_twitter_30.txt_filt")

dados <- data.frame()
for (arquivo in arquivos) {
  dados <- rbind(dados, data.frame(arquivo, ping = read.table(arquivo)[,1]))
}
dados %>%
  group_by(arquivo) %>%
  summarise(media_ping = mean(ping))
ggplot(dados, aes(x = arquivo, y = ping, col = arquivo)) +
  geom_jitter(height = 0)

res <- dados %>%
       group_by(arquivo) %>%
       summarise(media = mean(ping),
                 desvio_padrao = sd(ping),
                 tamanho_amostra = n(),
                 z = z_f(.95),
                 ic_erro = ic_erro_f(z, desvio_padrao, tamanho_amostra),
                 ic_menor = media - ic_erro,
                 ic_maior = media + ic_erro,
                 n_ideal = n_f(z, desvio_padrao, margem_erro = 10, media))
res
ggplot(res, aes(x = arquivo, y = media)) +
  geom_errorbar(aes(ymin = ic_menor, ymax = ic_maior)) +
  geom_point()
